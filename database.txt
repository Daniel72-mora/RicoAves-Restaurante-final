
CREATE DATABASE IF NOT EXISTS rico_aves;
USE rico_aves;


DROP TABLE IF EXISTS pedidos;
DROP TABLE IF EXISTS productos;
DROP TABLE IF EXISTS clientes;


CREATE TABLE clientes (
    id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    telefono VARCHAR(15),
    direccion VARCHAR(150),
    email VARCHAR(100)
);


CREATE TABLE productos (
    id_producto INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    descripcion TEXT,
    precio DECIMAL(10, 2) NOT NULL,
    categoria ENUM('Individual', 'Combo', 'Bebida') DEFAULT 'Individual'
);


CREATE TABLE pedidos (
    id_pedido INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT,
    id_producto INT,
    cantidad INT DEFAULT 1,
    fecha_pedido TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente) ON DELETE CASCADE,
    FOREIGN KEY (id_producto) REFERENCES productos(id_producto) ON DELETE CASCADE
);

Consultas Útiles (Eliminar y Reportar)
Para ver cuánto debe pagar un pedido total:

SELECT id_pedido, SUM(cantidad * precio_unitario) AS total_factura
FROM detalles_pedido
WHERE id_pedido = 101;

Para eliminar un producto específico de un pedido:

DELETE FROM detalles_pedido WHERE id_detalle = 5;


Para cancelar (eliminar) un pedido completo:

DELETE FROM pedidos WHERE id_pedido = 101;
